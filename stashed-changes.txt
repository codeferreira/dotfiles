+    local stats = vim.uv.fs_stat(path)
+    
+    if stats and stats.type == "directory" then
+      -- Delete the directory buffer
+      vim.api.nvim_buf_delete(args.buf, { force = true })
+      -- Open oil in the directory
+      require("oil").open(path)
+    end
+  end,
+})
